const WEB_APP_URL = "URL_DEPLOYMENT_GAS_KAMU"; // Link dari Deploy > Web App async function apiCall(data) { const response = await fetch(WEB_APP_URL, { method: "POST", body: JSON.stringify(data) }); return await response.json(); } // Contoh cara pakai di fungsi Login async function handleLogin() { const u = document.getElementById('username').value; const p = document.getElementById('password').value; showLoading(true); const isSuccess = await apiCall({ action: "login", user: u, pass: p }); showLoading(false); if(isSuccess) { showPage('admin'); loadDataAdmin(); } else { alert("Login Gagal!"); } }
